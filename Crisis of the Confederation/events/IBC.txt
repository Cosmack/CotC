#Intergalactic Banking Consortium
namespace = IBC

# IBC Expelled
character_event = {
	id = IBC.105
	desc = EVTDESC_IBC_105
	picture = GFX_evt_ibc_market
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = {
		name = EVTOPTA_IBC_105
		wealth = 200
		prestige = -100
		#banish_religion = ibc
		hidden_tooltip = {
			any_realm_lord = {
				limit = { 
					OR = {
						ai = no
						has_character_modifier = borrowed_from_ibc
					}
				}
				character_event = { id = IBC.107 }
			}
		}
	}
}

# IBC Expelled (Notifier for Vassals)
character_event = {
	id = IBC.107
	desc = EVTDESC_IBC_107
	picture = GFX_evt_ibc_market
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_IBC_107
		trigger = {
			NOT = { has_character_modifier = borrowed_from_ibc }
		}
	}
	option = {
		name = EVTOPTB_IBC_107
		trigger = {
			has_character_modifier = borrowed_from_ibc
		}
		remove_character_modifier = borrowed_from_ibc
	}
}
