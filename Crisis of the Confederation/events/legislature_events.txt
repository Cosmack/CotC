# PASS ANTI-CENTRALIZATION BILL
character_event = {
	id = laws.10000
	desc = "EVTDESCLAWS10000"
	picture = GFX_evt_speech
	is_triggered_only = yes
	
	# Liege has no regent
	option = {
		trigger = {
			NOT = {
				any_realm_character = {
					has_minor_title = title_regent
					same_liege = ROOT
				}
			}
		}
		name = "EVTOPTACONFIRM"
		liege = { character_event = { id = laws.10010 tooltip = "EVTTOOLTIPLAWS10010" } }
	}
	
	# Liege has a regent, who is in the faction and can pass the bill automatically
	option = {
		trigger = {
			any_realm_character = {
				has_minor_title = title_regent
				same_liege = ROOT
			}
		}
		name = "EVTOPTBLAWS10000"
		random_realm_character = {
			limit = {
				has_minor_title = title_regent
				same_liege = ROOT
			}
			character_event = { id = laws.10001 tooltip = "EVTTOOLTIPLAWS10001" }
		}
		clr_character_flag = flag_making_faction_ultimatum
	}
	
	# Nevermind
	option = {
		name = "EVTOPTNEVERMIND"
		clr_character_flag = flag_making_faction_ultimatum
		ai_chance = { factor = 0 }
	}
}

					religion = neo_feudal

					religion = solarian
				}
				trait = radical
			}
			
			# This weakens the realm either way, so situational bonuses do not apply
		}
	}
	
	option = {
		name = "EVTOPTREFUSE"
		# Starts a constitutional crisis
		add_character_modifier = { name = constitutional_crisis duration = -1 }
		add_character_modifier = { name = crisis_law_against_ruler duration = -1 }

		}

			limit = {
				NOT = { religion = terran_imperial }
				NOT = { religion = neo_feudal }
				NOT = { religion = astrist }
			}
			piety = -200
		}
		if = {
			limit = {
				OR = {
					religion = terran_federalist
					religion = terran_republican
					religion = colonial_separatist
				}
			}
			piety = -300
		}
		FROM = {
			character_event = { id = laws.407 tooltip = "EVTTOOLTIPLAWS407" } # Automatically oppose ruler
			opinion = { who = ROOT modifier = opinion_tyrant }
			any_faction_backer = {
				faction = faction_anti_centralization
				opinion = { who = ROOT modifier = opinion_tyrant }
				character_event = { id = laws.407 tooltip = "EVTTOOLTIPLAWS407" } # Automatically oppose ruler
			}
		}
		hidden_tooltip = {
			any_realm_character = {
				limit = { liege = { character = ROOT } }
				character_event = { id = laws.10012 } # Inform vassals
				character_event = { id = laws.500 days = 30 random = 15 } # AI takes sides for or against
			}
		}
	}
}
