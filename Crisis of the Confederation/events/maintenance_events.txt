# Wrong holding type management for the AI
character_event = { # Theocracies - these are particularly troubling
	id = maintenance.20
	hide_window = yes
	
	only_rulers = yes
	
	trigger = {
		ai = yes
		is_theocracy = no
		any_demesne_title = {
			can_be_given_away = yes
			tier = count
			is_theocracy = yes
		}
	}
	
	immediate = {
		# If no courtiers are available, make one up
		if = {
			limit = {
				NOT = {
					any_courtier = {
						is_landed = no
						NOT = { dynasty = ROOT }
						age = 16
						prisoner = no
						NOT = { trait = incapable }
					}
				}
			}
			capital_scope = {
				ROOT = {
					random_list = {
						50 = {
							create_random_priest = {
								random_traits = yes
								age = 30
								culture = PREV
								religion = PREV
								dynasty = culture
								female = yes
							}
						}
						50 = {
							create_random_priest = {
								random_traits = yes
								age = 30
								culture = PREV
								religion = PREV
								dynasty = culture
								female = no
							}
						}
					}
				}
			}
		}
		random_courtier = {
			limit = {
				is_landed = no
				NOT = { dynasty = ROOT }
				age = 16
				prisoner = no
				NOT = { trait = incapable }
			}
			ROOT = {
				random_demesne_title = {
					limit = {
						can_be_given_away = yes
						tier = count
						is_theocracy = yes
					}
					grant_title = PREVPREV
					remove_claim = ROOT
				}
			}
		}
		character_event = { id = maintenance.20 } # Recursive
	}
}

# Fixing titles with bad succession laws
character_event = {
	id = maintenance.100
	desc = "EVTDESCMAINTENANCE100"
	
	only_rulers = yes
	
	trigger = {
		is_feudal = yes
		is_patrician = no
		any_demesne_title = {
			NOT = { has_law = succ_appointment }
			crownlaw_title = {
				holder_scope = {
					NOT = { character = ROOT }
					is_vassal_or_below = ROOT
				}
				has_law = military_vassal_legitimacy_2
			}
			OR = {
				is_tribal_type_title = yes
				crownlaw_title = { has_law = centralization_2 }
				crownlaw_title = { has_law = centralization_3 }
				crownlaw_title = { has_law = centralization_4 }
			}
		}
	}
	
	option = {
		name = "EVTOPTAMAINTENANCE100"
		any_demesne_title = {
			limit = {
				NOT = { has_law = succ_appointment }
				crownlaw_title = {
					holder_scope = {
						NOT = { character = ROOT }
						is_vassal_or_below = ROOT
					}
					has_law = military_vassal_legitimacy_2
				}
				OR = {
					is_tribal_type_title = yes
					crownlaw_title = { has_law = centralization_2 }
					crownlaw_title = { has_law = centralization_3 }
					crownlaw_title = { has_law = centralization_4 }
				}
			}
			add_law = succ_appointment
		}
	}
}
