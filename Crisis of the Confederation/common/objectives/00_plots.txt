				}

			AND = {
				tier = DUKE

					AND = {
						NOT = {
							any_demesne_title = {
								tier = COUNT

							}
						}
						NOT = {
							any_vassal = {
								any_demesne_title = {
									tier = COUNT
									location = { duchy = { title = ROOT } }
								}
							}
						}
					}
				}
			}
			# Count titles in ruler's primary duchy
			AND = {
				tier = COUNT
				location = {
					duchy = {
						holder_scope = { character = FROM }
						FROM = { capital_scope = { duchy = { title = PREVPREVPREV } } }
					}
				}
			}
			# Count's secondary county
			AND = {
				tier = COUNT
				location = { is_capital = no }
				holder_scope = {
					lower_tier_than = DUKE
					num_of_count_titles = 2

# Get a character killed
plot_kill_character = {
	type = realm_characters_plus # All characters in the same realm, plus rivals and relatives of self, parents, children and spouse
	intrigue_plot = yes
	murder_plot = yes
	
	# Plotter scope
	potential = {
		prisoner = no
		age = 16
		NOT = { trait = incapable }
		NOT = {
			AND = {
				ai = yes
				trait = friendly
			}
		}
		
		hidden_tooltip = {
			OR = {
				ai = no # Human
				trait = ambitious
				trait = cold
				trait = lunatic
				trait = radical
				any_rival = { always = yes }
			}
		}
	}
	
	# Target allow trigger for when players target a specific character in the GUI
	
	player_allow = {
		NOT = { is_child_of = FROM } # Not my own children
		NOT = { any_spouse = { character = FROM } } # This case is covered by another plot
		is_within_diplo_range = FROM
	}
	
	creation_effect = {
		if = {
			limit = { FROM = { ai = yes } }
			FROM = {
				clr_character_flag = flag_ai_plot_timer
				set_character_flag = flag_ai_plot_timer
			}
		}
	}
	
	# Target character scope
	allow = {
		NOT = { reverse_opinion = { who = FROM value = 75 } } # Don't kill these people at all!
		NOT = { is_child_of = FROM } # Not my own children
		OR = {
			NOT = { character = FROM }  # Not against myself! (unless I am bonkers)
			trait = lunatic
		}
		NOT = { any_spouse = { character = FROM } } # This case is covered by another plot
		is_within_diplo_range = FROM
		FROM = {
			# List of possible motives
			OR = {
				# Inheritances
				any_heir_title = { holder_scope = { character = ROOT } }
				AND = {
					has_minor_title = title_expected_successor
					liege = { character = ROOT }
				}
				any_pretender_title = { current_heir = { character = ROOT } }
				spouse = { any_heir_title = { holder_scope = { character = ROOT } } }
				spouse = {
					has_minor_title = title_expected_successor
					liege = { character = ROOT }
				}
				spouse = { any_pretender_title = { current_heir = { character = ROOT } } }
				any_child = { any_heir_title = { holder_scope = { character = ROOT } } }
				any_child = {
					has_minor_title = title_expected_successor
					liege = { character = ROOT }
				}
				any_child = { any_pretender_title = { current_heir = { character = ROOT } } }
				
				# People whose jobs I want
				AND = {
					has_ambition = obj_become_chancellor
					liege = { job_chancellor = { character = ROOT } }
				}
				AND = {
					has_ambition = obj_become_marshal
					liege = { job_marshal = { character = ROOT } }
				}
				AND = {
					has_ambition = obj_become_treasurer
					liege = { job_treasurer = { character = ROOT } }
				}
				AND = {
					has_ambition = obj_become_spymaster
					liege = { job_spymaster = { character = ROOT } }
				}
				AND = {
					has_ambition = obj_become_spiritual
					liege = { job_spiritual = { character = ROOT } }
				}
				
				# People whose heirs I can press weak claims on
				ROOT = {
					has_regent = no
					any_demesne_title = {
						FROM = { has_weak_claim = PREV }
						current_heir = {
							OR = {
								NOT = { age = 16 }
								trait = incapable
							}
						}
					}
				}
				# Good decapitation strike choices
				any_current_enemy = {
					character = ROOT
					OR = {
						martial = 20
						ruled_years = 40
					}
				}
				# A liege I hate
				liege = {
					character = ROOT
					NOT = { reverse_opinion = { who = FROM value = -74 } }
				}
				# A rival
				any_rival = { character = ROOT }
			}
		}
	}
	
	chance = {
		factor = 2
		modifier = {
			factor = 0
			FROM = {
				ai = yes
				any_backed_character = {
					OR = {
						has_plot = plot_kill_character
						has_plot = plot_kill_spouse
					}
					plot_target_char = { character = ROOT }
				}
			}
		}
		
		# Personality
		modifier = {
			factor = 5
			FROM = { trait = cold }
		}
		
		modifier = {
			factor = 0.1
			FROM = {
				trait = humble
				NOT = {
					liege = {
						character = ROOT
						NOT = { reverse_opinion = { who = FROM value = -74 } }
					}
				}
				# A rival
				NOT = { any_rival = { character = ROOT } }
			}
		}
		modifier = {
			factor = 10
			FROM = { trait = ambitious }
		}
		
		# Ambitions
		
		
		# Moral authority
		modifier = {
			factor = 0.5
			piety = 0
			FROM = {
				religion = ROOT
				religion_authority = 0.75
			}

			piety = 0
			FROM = {

				religion_authority = 0.5
				NOT = { trait = pragmatic }
