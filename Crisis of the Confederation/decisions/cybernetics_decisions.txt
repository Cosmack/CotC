	open_cybernetics_menu_decision = {
		potential = {
			ai = no
			NOT = { has_character_flag = flag_cybernetics_menu_open }

	get_dni_decision = {
		is_high_prio = yes
		potential = {
			NOT = { trait = dni }
			age = 16
			NOT = { trait = incapable }
			prisoner = no
			OR = {
				ai = yes
				has_character_flag = flag_cybernetics_menu_open
			}
		}
		allow = {
			wealth = 100
			OR = {
				NOT = { religion_group = religious_group }
				trait = pragmatic
				piety = 100
			}
		}
		effect = {
			wealth = -100
			add_trait = dni
			hidden_tooltip = { character_event = { id = cybernetics.5000 } }
			if = {
				limit = {
					is_landed = yes
					religion_group = religious_group
				}
				religion_authority = { modifier = ruler_cybernetics_bad years = 10 }
				piety = -100
			}
			if = {
				limit = {
					is_landed = yes
					religion_group = transhumanist_group
					NOT = { religion = bio_directionist }
				}
				religion_authority = { modifier = ruler_cybernetics_good years = 10 }
				piety = 100
			}
		}
		ai_will_do = {
			factor = 1
			# No reason to get a DNI
			modifier = {
				factor = 0
				NOT = { has_ambition = obj_become_marshal }
				NOT = { has_ambition = obj_become_chancellor }
				NOT = { has_ambition = obj_become_spymaster }
				NOT = { has_ambition = obj_become_treasurer }
				NOT = { has_ambition = obj_become_spiritual }
				NOT = { has_ambition = obj_improve_martial }
				NOT = { has_ambition = obj_improve_diplomacy }
				NOT = { has_ambition = obj_improve_stewardship }
				NOT = { has_ambition = obj_improve_intrigue }
				NOT = { has_ambition = obj_improve_learning }
				NOT = { religion = cyberneticist }
				NOT = { religion = machine_cultist }
				NOT = { religion = hiver }
				NOT = { trait = open_minded }
				NOT = { trait = ambitious }
			}
			
			# Can't afford one
			modifier = {
				factor = 0
				NOT = { wealth = 200 }
				OR = {
					NOT = { monthly_income = 0 }
					has_ambition = obj_amass_wealth
				}
			}
			
			# Ideological concerns
			modifier = {
				factor = 0
				religion_group = religious_group
				NOT = { trait = pragmatic }
			}
			
			# Just plain squeamish
			modifier = {
				factor = 0
				OR = {
					trait = close_minded
					trait = spiritual
				}
				NOT = { trait = open_minded }
				NOT = { religion = cyberneticist }
				NOT = { religion = machine_cultist }
				NOT = { religion = hiver }
			}
		}
